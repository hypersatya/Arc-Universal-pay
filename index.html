const USDC_ADDR = "0x3600000000000000000000000000000000000000"; const ARC_CHAIN_ID = "0x4cef52"; const INR_RATE = 83.50; let userAddress = ""; let provider, signer; // 1. CONNECT & DASHBOARD LOAD async function connectWallet() { if (!window.ethereum) return alert("Install OKX Wallet!"); try { await window.ethereum.request({ method: 'wallet_switchEthereumChain', params: [{ chainId: ARC_CHAIN_ID }], }).catch(async (err) => { if (err.code === 4902) { await window.ethereum.request({ method: 'wallet_addEthereumChain', params: [{ chainId: ARC_CHAIN_ID, chainName: "Arc Testnet", nativeCurrency: { name: "USDC", symbol: "USDC", decimals: 18 }, rpcUrls: ["https://rpc.testnet.arc.network/"], blockExplorerUrls: ["https://testnet.arcscan.app/"] }] }); } }); const accounts = await window.ethereum.request({ method: "eth_requestAccounts" }); userAddress = accounts[0]; provider = new ethers.providers.Web3Provider(window.ethereum); signer = provider.getSigner(); document.getElementById("loginScreen").classList.add("hidden"); document.getElementById("dashboard").classList.remove("hidden"); document.getElementById("walletAddr").innerText = userAddress.slice(0, 6) + "..." + userAddress.slice(-5).toUpperCase(); fetchBalance(); loadHistory(); } catch (e) { console.error(e); } } // 2. RECEIVE FEATURE function openReceive() { document.getElementById("receiveModal").classList.remove("hidden"); document.getElementById("fullAddrDisplay").innerText = userAddress; document.getElementById("qrcode").innerHTML = ""; new QRCode(document.getElementById("qrcode"), { text: userAddress, width: 180, height: 180, colorDark: "#030613", colorLight: "#ffffff" }); } function copyAddr() { navigator.clipboard.writeText(userAddress); alert("Address Copied! ✅"); } // 3. TRANSACTION HISTORY LOGIC function saveTx(to, amount, hash) { let history = JSON.parse(localStorage.getItem("arc_tx_history") || "[]"); history.unshift({ to, amount, hash, time: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) }); localStorage.setItem("arc_tx_history", JSON.stringify(history.slice(0, 5))); loadHistory(); } function loadHistory() { const list = document.getElementById("txHistoryList"); let history = JSON.parse(localStorage.getItem("arc_tx_history") || "[]"); if (history.length === 0) { list.innerHTML = `
No recent transactions
`; return; } list.innerHTML = history.map(tx => `

To: ${tx.to.slice(0,6)}...${tx.to.slice(-4)}

${tx.time}

-${tx.amount} USDC

Details
`).join(''); } // 4. SEND PAYMENT (FIXED FOR OKX) async function executePayment() { const to = document.getElementById("toAddress").value; const amount = document.getElementById("amount").value; const btn = document.getElementById("finalSendBtn"); if (!ethers.utils.isAddress(to) || !amount) return alert("Sahi details dalo!"); try { btn.innerText = "SENDING..."; btn.disabled = true; const abi = ["function transfer(address, uint256) returns (bool)"]; const contract = new ethers.Contract(USDC_ADDR, abi, signer); const tx = await contract.transfer(to, ethers.utils.parseUnits(amount, 6), { gasLimit: 120000, type: 0 // Legacy mode for OKX stability }); await tx.wait(); saveTx(to, amount, tx.hash); // Save to history alert("Success! ✅"); closeModal('transferModal'); fetchBalance(); } catch (e) { alert("Transaction Failed! Faucet se USDC claim karo."); } finally { btn.innerText = "CONFIRM SEND"; btn.disabled = false; } } // HELPERS async function fetchBalance() { try { const abi = ["function balanceOf(address) view returns (uint256)"]; const contract = new ethers.Contract(USDC_ADDR, abi, provider); const bal = await contract.balanceOf(userAddress); const formatted = ethers.utils.formatUnits(bal, 6); document.getElementById("usdcBal").innerText = parseFloat(formatted).toFixed(2); document.getElementById("inrBal").innerText = (formatted * INR_RATE).toLocaleString('en-IN'); } catch (e) { document.getElementById("usdcBal").innerText = "0.00"; } } function openTransfer() { document.getElementById("transferModal").classList.remove("hidden"); } function closeModal(id) { document.getElementById(id).classList.add("hidden"); } function updateInrPreview(v) { document.getElementById("previewInr").innerText = (v * INR_RATE).toLocaleString('en-IN'); }
